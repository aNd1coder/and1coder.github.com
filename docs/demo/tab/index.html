参数说明

                $.fn.tab.defaults = {
                    trigger:".ui-tab-trigger", //触发点容器
                    container:".ui-tab-container", //内容容器
                    current:"current", //当前触发点以及内容节点样式
                    triggerType:"mouseover", //触发事件类型
                    selectedIndex:0, //默认被选中的索引
                    auto:false, //是否自动切换
                    interval:2000, //时间间隔
                    lazyload:true //是否延迟加载
                }
